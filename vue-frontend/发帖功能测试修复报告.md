2. **02-04**: 点赞测试失败 - `button[aria-label*="点赞"]`选择器超时
3. **02-05**: 评论测试失败 - `.post-card`选择器超时

## 🔍 根本原因分析

### 问题1：发帖模态框选择器错误

**测试代码使用的选择器**：
```javascript
const textarea = this.page.locator('textarea[placeholder*="分享"]')
```

**实际前端实现**：
```vue
<!-- PostModal.vue -->
<textarea 
  v-model="form.content" 
  :placeholder="currentPlaceholder"  <!-- 动态占位符 -->
  maxlength="2000"
  class="w-full min-h-[140px] text-base border-none bg-transparent resize-none p-4 pr-10 focus:outline-none placeholder:text-gray-400">
</textarea>
```

**占位符列表**（随机选择）：
```javascript
const placeholders = [
  '今天想说点什么？✨',
  '分享你的校园故事...',
  '有什么新鲜事想告诉大家？',
  '写下此刻的心情吧~',
  '记录美好瞬间 📸',
  '想对TA说些什么？💕'
]
```

**问题**：
- 占位符是随机的，不是固定的"分享"
- 测试代码中使用`placeholder*="分享"`无法匹配所有情况
- 只有当随机到"分享你的校园故事..."时才能匹配成功

### 问题2：帖子卡片和按钮选择器错误

**测试代码使用的选择器**：
```javascript
// 点赞按钮
page.locator('button[aria-label*="点赞"]')

// 分享按钮
page.locator('button[aria-label*="分享"]')

// 帖子卡片
page.locator('.post-card')
```

**实际前端实现**：
```vue
<!-- Home.vue -->
<Card v-for="post in posts" :key="post.id"
  class="border-none shadow-sm hover:shadow-md transition-all rounded-2xl bg-white overflow-hidden cursor-pointer"
  @click="router.push(`/post/${post.id}`)">
  
  <!-- 点赞按钮 - 没有aria-label -->
  <Button variant="ghost" 
    class="w-full rounded-xl text-gray-500 hover:text-pink-500 hover:bg-pink-50"
    @click.stop="handleLike(post)">
    <Heart class="w-4 h-4 mr-1.5" />
    <span class="text-xs font-medium">{{ post.likeCount || 0 }}</span>
  </Button>
  
  <!-- 评论按钮 - 没有aria-label -->
  <Button variant="ghost" 
    class="w-full rounded-xl text-gray-500 hover:text-blue-500 hover:bg-blue-50">
    <MessageCircle class="w-4 h-4 mr-1.5" />
    <span class="text-xs font-medium">{{ post.commentCount || 0 }}</span>
  </Button>
  
  <!-- 分享按钮 - 没有aria-label -->
  <Button variant="ghost"
    class="w-full rounded-xl text-gray-500 hover:text-[rgb(33,111,85)] hover:bg-green-50"
    @click.stop="handleShare(post)">
    <Share2 class="w-4 h-4 mr-1.5" />
    <span class="text-xs font-medium">{{ post.shareCount || 0 }}</span>
  </Button>
</Card>
```

**问题**：
- 帖子卡片使用的是`<Card>`组件，不是`.post-card` class
- 点赞、评论、分享按钮都没有`aria-label`属性
- 需要使用图标或文本内容来定位

**修复后**：
```javascript
async createPost(content, category = '表白') {
  await this.openPostModal()
  
  // 等待模态框出现（通过DialogContent）
  await this.page.waitForSelector('[role="dialog"]', { timeout: 5000 })
  await this.page.waitForTimeout(500)
  
  // 使用更通用的选择器定位textarea（不依赖placeholder）
  const textarea = this.page.locator('textarea').first()
  await textarea.waitFor({ state: 'visible', timeout: 5000 })
  await textarea.fill(content)
  await this.page.waitForTimeout(300)
  
  // 选择分类（点击分类按钮）
  const categoryButton = this.page.locator('button').filter({ hasText: category })
  if (await categoryButton.isVisible().catch(() => false)) {
    await categoryButton.click()
    await this.page.waitForTimeout(300)
  }
  
  // 点击发布按钮（在模态框底部）
  const publishButton = this.page.locator('[role="dialog"] button:has-text("发 布")')
  await publishButton.waitFor({ state: 'visible', timeout: 5000 })
  await publishButton.click()
  
  // 等待发布成功提示
  await this.page.waitForSelector('text=发布成功', { timeout: 10000 })
  await this.page.waitForTimeout(1000)
}
```

#### 新增帖子交互方法

在`HomePage` class中添加：

```javascript
// 点赞帖子（通过Heart图标）
async likePost(index = 0) {
  // 等待帖子加载
  await this.page.waitForTimeout(2000)
  
  // 找到所有点赞按钮（包含Heart图标的按钮）
  const likeButtons = this.page.locator('button').filter({ 
    has: this.page.locator('svg').filter({ hasText: '' }) // Heart图标
  })
  
  // 点击指定索引的点赞按钮
  await likeButtons.nth(index).click()
  await this.page.waitForTimeout(500)
}

// 进入帖子详情
async openPost(index = 0) {
  // 等待帖子加载
  await this.page.waitForTimeout(2000)
  
  // 点击帖子卡片（Card组件）
  const postCards = this.page.locator('[class*="rounded-2xl"][class*="bg-white"]').filter({
    has: this.page.locator('text=/表白|寻物|闲置|吐槽|活动|求助|其他/')
  })
  
  await postCards.nth(index).click()
  
  // 等待详情页加载
  await this.page.waitForURL(/\/post\/\d+/, { timeout: 10000 })
}

// 分享帖子
async sharePost(index = 0) {
  // 等待帖子加载
  await this.page.waitForTimeout(2000)
  
  // 找到所有分享按钮（包含Share2图标的按钮）
  const shareButtons = this.page.locator('button').filter({
    has: this.page.locator('svg') // Share2图标
  }).filter({ hasText: /^\d+$/ }) // 包含数字的按钮
  
  // 点击第三组按钮（点赞、评论、分享）
  const allActionButtons = this.page.locator('button[class*="rounded-xl"]')
  const shareButton = allActionButtons.filter({
    has: this.page.locator('svg')
  }).nth(index * 3 + 2) // 每个帖子有3个按钮，分享是第3个
  
  await shareButton.click()
  await this.page.waitForTimeout(500)
}
```

### 更新page-objects.js

```javascript
export class HomePage {
  constructor(page) {
    this.page = page
  }

  async goto() {
    await this.page.goto('/home')
    await this.page.waitForLoadState('networkidle')
  }

  async openPostModal() {
    // 点击顶部或侧边栏的发布按钮
    const publishButton = this.page.locator('button').filter({ hasText: /发布|发帖/ }).first()
    await publishButton.click()
    await this.page.waitForTimeout(1000)
  }

  async createPost(content, category = '表白') {
    await this.openPostModal()
    
    // 等待对话框出现
    await this.page.waitForSelector('[role="dialog"]', { timeout: 5000 })
    await this.page.waitForTimeout(500)
    
    // 填写内容（使用第一个textarea）
    const textarea = this.page.locator('[role="dialog"] textarea').first()
    await textarea.waitFor({ state: 'visible', timeout: 5000 })
    await textarea.fill(content)
    await this.page.waitForTimeout(300)
    
    // 选择分类
    const categoryButton = this.page.locator('[role="dialog"] button').filter({ hasText: category })
    if (await categoryButton.isVisible().catch(() => false)) {
      await categoryButton.click()
      await this.page.waitForTimeout(300)
    }
    
    // 点击发布按钮
    const publishButton = this.page.locator('[role="dialog"] button:has-text("发 布")')
    await publishButton.waitFor({ state: 'visible', timeout: 5000 })
    await publishButton.click()
    
    // 等待发布成功
    await this.page.waitForSelector('text=发布成功', { timeout: 10000 })
    await this.page.waitForTimeout(1000)
  }

  async likePost(index = 0) {
    await this.page.waitForTimeout(2000)
    
    // 找到所有帖子的操作按钮组
    const actionButtons = this.page.locator('button[class*="rounded-xl"]').filter({
      has: this.page.locator('svg')
    })
    
    // 点赞按钮是每组的第1个（index * 3 + 0）
    await actionButtons.nth(index * 3).click()
    await this.page.waitForTimeout(500)
  }

  async openPost(index = 0) {
    await this.page.waitForTimeout(2000)
    
    // 点击帖子卡片
    const postCards = this.page.locator('[class*="rounded-2xl"][class*="bg-white"][class*="cursor-pointer"]')
    await postCards.nth(index).click()
    
    // 等待详情页
    await this.page.waitForURL(/\/post\/\d+/, { timeout: 10000 })
  }

  async sharePost(index = 0) {
    await this.page.waitForTimeout(2000)
    
    // 找到所有帖子的操作按钮组
    const actionButtons = this.page.locator('button[class*="rounded-xl"]').filter({
      has: this.page.locator('svg')
    })
    
    // 分享按钮是每组的第3个（index * 3 + 2）
    await actionButtons.nth(index * 3 + 2).click()
    await this.page.waitForTimeout(500)
  }

  async searchPosts(keyword) {
    await this.page.fill('input[placeholder*="搜索"]', keyword)
    await this.page.press('input[placeholder*="搜索"]', 'Enter')
    await this.page.waitForTimeout(1000)
  }

  async filterByCategory(category) {
    await this.page.click(`button:has-text("${category}")`)
    await this.page.waitForTimeout(1000)
  }
}
```

## 💡 经验教训

1. **不要依赖动态内容**：占位符、随机文本等不适合作为选择器
2. **优先使用结构化选择器**：role、data-testid优于class和文本
3. **测试代码应适应前端变化**：而不是让前端适应测试
4. **使用索引定位时要小心**：确保DOM结构稳定
5. **添加充足的等待时间**：避免因加载延迟导致的误报
