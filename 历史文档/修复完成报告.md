    {

    }
  );

  // 错误处理
  if (e.name === "AbortError") {
    appStore.showToast("已取消AI润色", "info");

    appStore.showToast("AI 开小差了，请稍后再试 🤖", "warning");
  }
};

const cancelPolish = () => {
  if (abortController.value) {
